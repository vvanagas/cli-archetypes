selection_rule:
  meaning: Controls are trigger-bound obligations, not optional style advice.
  source_of_truth:
    obligation: this file (trigger, floor, ceiling, required, forbidden, proof)
    attachment: server-archetype-control-map.txt (default / conditional / forbidden)
  apply_order:
    - classify_archetype
    - detect_triggers
    - attach_default_controls
    - attach_conditional_controls
    - drop_control_if_floor_condition_holds
    - reject_forbidden_controls_unless_reclassified_or_justified
    - prove_each_attached_control

controls:
  - id: C01
    name: schema-at-boundary
    trigger: input crosses a trust boundary
    floor: in-process call from already-validated code; no boundary is crossed
    ceiling: published-contract schema registry + versioning when external consumers depend on the shape (mode = consumer breakage, not local trust)
    use_when:
      - HTTP body/query/params
      - CLI argv/stdin
      - config file
      - environment variables
      - webhook payload
      - queue message
      - subprocess output
      - LLM output
      - external API response
      - realtime frame/message (EVERY frame, not just the handshake)
    required:
      - runtime schema exists
      - raw input treated as unknown
      - validation happens before service/domain code
      - validated DTO passed inward
      - type derived from schema where stack supports it
    forbidden:
      - JSON.parse_as_T
      - raw_any_into_service_or_domain
      - duplicated_schema_and_type
      - validation_hidden_inside_domain_logic
      - trusting_LLM_or_subprocess_output_without_validation
      - post_handshake_frames_treated_as_trusted
    proof:
      auto:
        - no_any
        - no_unchecked_cast
      tests:
        - valid_input_passes
        - missing_required_field_fails
        - wrong_type_fails
        - hostile_extra_field_handled

  - id: C02
    name: thin-handler
    trigger: request, job, command, or event reaches an entrypoint
    floor: none — always applies
    ceiling: command bus / mediator when many command types need dynamic dispatch (mode = dispatch fan-out, not handler length)
    required:
      - receive typed input
      - read already-established context
      - call one use case or query path
      - map Result/Error to response/status/ack/exit
    forbidden:
      - business_logic_in_handler
      - SQL_in_handler
      - external_IO_in_handler
      - dependency_construction_in_handler
      - transaction_ownership_in_handler_for_real_commands
      - multiple_hidden_use_cases_in_one_handler
    proof:
      review:
        - one_use_case_call
        - no_domain_branching
        - no_DB_import
        - no_external_SDK_import

  - id: C03
    name: domain-pure-function
    trigger: business rule, invariant, calculation, or state transition exists
    floor: pure store/retrieve with no rule — there is no domain to extract
    ceiling: none
    required:
      - pure function or pure domain module
      - plain input and output data
      - no framework imports
      - no DB, network, filesystem, queue, clock, env, or subprocess IO
    forbidden:
      - domain_reads_environment
      - domain_calls_database
      - domain_calls_external_API
      - domain_mutates_global_state
      - domain_imports_transport_or_persistence_types
    proof:
      tests:
        - pure_rule_unit_tests
        - deterministic_same_input_same_output
        - edge_cases_covered

  - id: C04
    name: result-error-union
    trigger: recoverable failure must be handled by caller
    floor: programmer errors and invariants still throw — do not Result-ify a bug
    ceiling: none
    required:
      - Result_ok_or_err_or_stack_native_equivalent
      - typed error variants
      - caller branches explicitly
      - variants map exhaustively at boundary
    forbidden:
      - throwing_for_domain_rejection
      - null_or_undefined_as_error
      - bare_error_string
      - catch_all_then_generic_failure
    proof:
      auto:
        - exhaustive_switch_where_supported
      tests:
        - each_error_variant_mapped
        - business_error_not_retried
        - transient_error_classified

  - id: C05
    name: exhaustive-status-mapping
    trigger: errors leave service boundary
    floor: single outcome only (one success, no distinct error classes)
    ceiling: none
    required:
      - one mapping table or one mapping function
      - stable error codes
      - business/client/transient/programmer failures are distinct
      - HTTP status, CLI exit code, or queue decision is explicit
      - parse/validation failure on a consumed message routes to DLQ or explicit reject — a typed, terminal outcome (C39 merge)
    forbidden:
      - 200_with_error_body
      - generic_500_for_client_error
      - all_cli_failures_exit_1
      - ack_on_unknown_job_error
      - leaking_stack_trace_to_client
      - unparseable_message_scheduled_for_retry   # C39 merge — a poison pill retried is a worker crash loop; malformed is TERMINAL, not transient
    proof:
      tests:
        - validation_maps
        - not_found_maps
        - conflict_maps
        - forbidden_maps
        - transient_maps
        - unknown_maps_safely
        - malformed_message_routed_to_dlq_without_retry

  - id: C06
    name: timeout-every-IO
    trigger: external IO call exists
    floor: in-memory only, no external IO; OR the child is a supervised long-lived process whose lifetime IS the point (a watched dev server, a REPL, an interactive editor under a daemon/supervisor) — an unbounded wait is correct there and the obligation redirects to C27 graceful teardown, not a deadline
    ceiling: circuit breaker + bulkhead when repeated failures must shed load (mode = cascading failure, not one slow call)
    required:
      - connect timeout or stack equivalent
      - read/operation timeout or stack equivalent
      - timeout budget named
      - timeout maps to typed failure
    forbidden:
      - default_infinite_timeout
      - unbounded_subprocess_wait
      - request_timeout_greater_than_caller_budget
      - timeout_error_lost_as_generic_failure
    proof:
      tests:
        - timeout_maps_to_typed_error
        - caller_receives_expected_status_or_exit

  - id: C07
    name: retry-only-idempotent
    trigger: retry is proposed
    floor: operation naturally idempotent and cheap to fail — just fail and surface
    ceiling: durable retry queue / DLQ when retries must survive process death (mode = durability, not transient recovery)
    required:
      - operation classified as idempotent or guarded by idempotency key
      - retryable error classes listed
      - max attempts defined
      - backoff uses full jitter
      - business rejection is terminal
      - each attempt and the terminal outcome logged with the correlation id (C16)
    forbidden:
      - retry_non_idempotent_write
      - retry_business_rejection
      - unbounded_retry
      - retry_attempts_or_terminal_outcome_unlogged
    proof:
      tests:
        - retry_transient_error
        - no_retry_terminal_error
        - max_attempts_enforced

  - id: C08
    name: idempotency-key
    trigger: client, queue, webhook sender, or process may repeat a mutation
    floor: operation is naturally idempotent (e.g. PUT of full state)
    ceiling: C11 outbox / saga when the effect crosses a service boundary and cannot share one local transaction
    required:
      - stable key captured at boundary
      - same key produces one effect
      - same key returns stored or equivalent result
      - key claim stored atomically with effect where applicable
    forbidden:
      - retryable_POST_without_key
      - idempotency_checked_outside_effect_tx
      - key_not_in_unique_constraint
      - different_response_on_replay_without_reason
    proof:
      tests:
        - same_key_twice_one_effect
        - same_key_same_response
        - concurrent_same_key_one_winner

  - id: C09
    name: short-transaction
    trigger: DB write transaction exists
    floor: no DB write — no transaction
    ceiling: saga when the unit of work spans services and cannot share one transaction
    required:
      - transaction scope named
      - DB work only inside transaction
      - commit or rollback explicit
      - no slow external boundary inside transaction
    forbidden:
      - network_IO_inside_transaction
      - filesystem_or_subprocess_inside_transaction_when_slow
      - long_transaction_over_external_call
      - hidden_implicit_transaction_for_critical_command
    proof:
      review:
        - no_external_call_inside_tx
        - tx_boundary_visible
      tests:
        - mid_tx_failure_rolls_back

  - id: C10
    name: load-lock-decide-write
    trigger: mutation depends on current DB state
    floor: single-writer or no real contention — no lock needed; if the endpoint is a plain row update with no invariant, reclassify to http_crud (write) rather than attach C10
    ceiling: distributed lock / lease when one DB transaction cannot cover multi-node coordination
    required:
      - open transaction
      - load fresh state
      - lock row or use serializable retry
      - domain decision runs on loaded state
      - write effect in same transaction
    forbidden:
      - preload_entity_before_transaction
      - stale_read_then_write
      - stateful_validation_before_lock
      - decision_based_on_cached_state
    proof:
      tests:
        - concurrent_mutations_preserve_invariant
        - lost_update_prevented

  - id: C11
    name: outbox-after-state-change
    trigger: external call, event, or message follows DB state change
    floor: external call is idempotent and order-tolerant — order it (commit, then call with C08 key) instead of an outbox
    ceiling: saga when a later step must COMPENSATE (undo) the external effect; outbox guarantees delivery, not undo
    required:
      - state change and outbox row committed in same transaction
      - separate publisher dispatches outbox
      - publisher is replay-safe
      - receiver or downstream operation is idempotent
    forbidden:
      - HTTP_call_inside_transaction
      - publish_after_commit_without_outbox
      - DB_change_plus_external_call_inline
      - mark_sent_before_external_ack
    proof:
      tests:
        - state_and_outbox_commit_together
        - publisher_replay_safe
        - crash_after_commit_before_publish_recovers

  - id: C12
    name: stdout-data-stderr-logs
    trigger: CLI emits machine-consumable output
    floor: human-only tool with no machine-consumed output
    ceiling: none
    required:
      - stdout contains only data
      - stderr contains logs, progress, warnings
      - JSON output mode is parseable
      - human progress does not corrupt stdout
    forbidden:
      - logs_on_stdout
      - progress_mixed_with_JSON
      - warnings_before_JSON_on_stdout
    proof:
      tests:
        - stdout_parseable

  - id: C13
    name: semantic-exit-code
    trigger: CLI outcome has multiple meanings
    floor: single binary outcome — 0 success / 1 failure suffices
    ceiling: none
    required:

    forbidden:

  - id: C14
    name: raw-body-before-webhook-parse

    floor: internal/trusted caller with no signature — but then question whether it is a webhook at all
    ceiling: mTLS / mutual auth when per-request signature is insufficient (mode = channel-level trust)

    forbidden:

  - id: C15
    name: ownership-authz-on-fresh-state

    floor: authz is a static role/scope check independent of resource state — a coarse edge check is enough
    ceiling: external policy engine when authz rules are many and centrally managed

    forbidden:

      - the C16 context's cancellation token reaches every network/DB client — C16 CARRIES it, this card makes clients HONOR it
      - downstream query/connection physically aborted when the token fires
      - point of no return respected — once the transaction commits (or C11's floor path has begun its commit-then-call), cancellation is IGNORED; partial-effect prevention outranks resource reclaim (C11's outbox makes this atomic by construction)
    forbidden:
      - orphaned_query_running_after_client_disconnect   # the bored user refreshes 10 times = 10 expensive queries for nobody, starving real traffic
      - swallowing_cancellation_to_force_completion_on_the_read_path
      - honoring_cancellation_between_commit_and_durable_handoff
    proof:
      tests:
        - client_disconnect_aborts_downstream_query
        - cancellation_after_commit_does_not_tear_the_effect

  - id: C57
    name: ephemeral-disk-cleanup-guarantee
    trigger: request or job stages data on local disk (temp files for transforms, uploads, renders, exports)
    floor: pure in-memory streaming — nothing touches the filesystem
    ceiling: per-invocation ephemeral filesystems (Lambda-class) destroyed by the platform (mode = platform-owned lifecycle)
    required:
      - temp files created under the OS temp directory with unique names
      - deletion guaranteed by finalizer (defer/finally) on EVERY exit path — success, domain error, timeout, panic
      - concurrent staged volume bounded — max staged bytes named (C31's disk cousin)
    forbidden:
      - temp_file_leaked_on_handled_domain_error    # the early 400 return that skips cleanup; weeks later DiskPressure evicts healthy pods
      - cleanup_delegated_to_external_cron
      - unbounded_concurrent_staging
    proof:
      tests:
        - domain_error_mid_processing_still_unlinks_temp_file
        - timeout_path_unlinks_temp_file

  - id: C58
    name: mandatory-pagination-bound
    trigger: query targets a collection whose size grows with user activity or time, OUTSIDE a C23-governed list endpoint (workers, jobs, internal reads, admin paths)
    floor: fixed-size-by-design tables (config, enums, US states); list endpoints whose C23 max limit is already enforced at the driver
    ceiling: streaming drivers / map-reduce pipelines explicitly built for unbounded extraction (mode = batch pipeline, not one query)
    required:
      - hard numeric LIMIT at the SQL/driver level — the bound lives where the bytes are produced, not in app-code hope

      - relation to C23 recorded — the edge envelope's max limit COMPILES DOWN to this; C23 without a driver-level bound is a promise without a mechanism
    forbidden:
      - select_star_into_process_memory_from_a_growing_table   # works in dev, works in prod for a year, OOMKills on the 10000th user
      - loop_over_unbounded_result_set_without_streaming
    proof:
      tests:
        - unbounded_internal_query_enforces_system_hard_limit
        - full_scan_path_batches_with_bounded_memory
